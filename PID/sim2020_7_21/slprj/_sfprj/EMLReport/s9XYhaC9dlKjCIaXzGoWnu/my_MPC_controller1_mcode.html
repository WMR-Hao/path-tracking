  4% u_out=[vel_l ; vel_r]
  5% ref_vel=[xr yr alphar vr omegar]
  6% e = [ex ey ealpha]
  7
  8%测试是否能得到
  9  statu=[2 -2 pi/2 0 0 0]
 10  u_out=[0; 0]
 11  ref_pos=[0;0;pi/4]; ref_vel=[2 0 0.5];  
 12%% 底盘参数
 13d = 0.18;
 14l = 0.16;
 15m = 24;
 16Iz = 11.89;
 17c = 513.6; %轮胎侧偏刚度
 18
 19%% 状态量                                                                           
 20x = statu(1);   y = statu(2);  alpha = statu(3);
 21vx = statu(4);  vy = statu(5);    omega = statu(6);
 22
 23Nc= 2; Np=4; %预测时域Np	控制时域Nc
 24
 25T=0.05;                                       
 26                                                                               
 27Xr=ref_pos(1);  Yr=ref_pos(2);  ALPHAr=ref_pos(3);
 28vxr=ref_vel(1); vyr=ref_vel(2); omegar=ref_vel(3);
 29                                                                                      
 30%控制量个数Nu
 31Nu=size(u_out,1);
 32
 33%状态量个数Nx
 34statu_error=[x-Xr;   y-Yr;   alpha-ALPHAr;  vx-vxr; vy-vyr; omega-omegar]; 
 35Nx=size(statu_error,1);
 36
 37% 控制矩阵求解(将连续矩阵转换为离散矩阵，采用近似算法  A=I+T*A(t),B=T*B(t))
 38% 此处：delta_X=AX+BU ————X=[a-r]=[X_I-Xr;Y_I-Yr;ALPHA_I-ALPHAr]statu_e=A*statu_e+B*u_e   %离散后的模型
 39    % OUT=delta_X=a-r
 40A=[ 1, 0, (-vx*sin(alpha)-vy*cos(alpha))*T, T*cos(alpha), T*sin(alpha), 0;
 41    0, 1, (vx*cos(alpha)-vy*sin(alpha))*T,  T*sin(alpha), T*cos(alpha), 0;
 42    0, 0, 1, 0, 0, T;
 43    0, 0, 0, 1, 0, 0;
 44    0, 0, 0, T*(2*c*(vy+l*omega))/(m*vx^2+0.0001), (-2*c*T)/(m*vx+0.0001)+1, (-2*c*l*T)/(m*vx+0.0001);
 45    0, 0, 0, -T*(2*c*l*(vy+l*omega))/(Iz*vx^2+0.0001), 2*c*l*T/(Iz*vx+0.0001), (2*c*l*l*T)/(vx*Iz+0.0001)+1;]  %+0.0001是为了避免分母为零
 46
 47B=[0 0;
 48   0 0;
 49   0 0;
 50   T/m T/m;
 51   0 0;
 52   -T*d/Iz T*d/Iz;]
 53
 54C = eye(Nx);
 55
 56Error = C * statu_error; % 误差
 57
 58e_pos=Error(1:Nx/2); %位置误差
 59e_vel=Error(Nx/2:Nx); %速度误差
 60
 61% 得到新的X=A_new*X+B_new*delta_u   X=[X;u-ur]
 62A_new = [A B;zeros(Nu,Nx) eye(Nu)];
 63B_new = [B;eye(Nu)];
 64C_new = [C,zeros(Nx,Nu)];
 65%% 
 66ur=[0; 0];
 67
 68statu_new = [Error;u_out-ur];
 69
 70PHI=zeros(Np*size(C_new*A_new,1),size(A_new,2));
 71for i=1:1:Np
 72    PHI( Nx*i-(Nx-1):Nx*i  ,:)=C_new*A_new^i;
 73end
 74
 75% THETA=zeros(Np*size(C_new*B_new,1),(Nc+1)*size(C_new*A_new*B_new ,2));
 76THETA=zeros(Np*size(C_new*B_new,1), Nc*size(C_new*A_new*B_new ,2))
 77for i=1:1:Np
 78%     for j=1:1:Nc+1
 79    for j=1:1:Nc
 80        if j<= i
 81            THETA( Nx*i-5:Nx*i,Nu*j-1:Nu*j )= C_new*A_new^(i-j)*B_new;
 82        else
 83            THETA( Nx*i-5:Nx*i,Nu*j-1:Nu*j ) = zeros( Nx,Nu );
 84        end
 85    end
 86end
 87
 88q = [1 0 0 0 0 0;
 89     0 1 0 0 0 0;
 90     0 0 1 0 0 0;
 91     0 0 0 1 0 0;
 92     0 0 0 0 1 0;
 93     0 0 0 0 0 1;];
 94Q=zeros(Nx*Np,Nx*Np);
 95for i=1:Np
 96    for j=1:Np
 97        if i==j
 98            Q(Nx*(i-1)+1:Nx*i , Nx*(i-1)+1:Nx*i) = q;
 99        end
100    end
101end
102
103% R = 1*eye( (Nc+1)*Nu,(Nc+1)*Nu);
104R = 1*eye( (Nc)*Nu,(Nc)*Nu);
105
106%% 二次型
107p=3;
108% H = [THETA'*Q*THETA+R zeros((Nc+1)*Nu,1);zeros(1,(Nc+1)*Nu) p];
109
110H = [THETA'*Q*THETA+R zeros(Nc*Nu,1);zeros(1,Nc*Nu) p];
111
112
113ERROR =PHI*statu_new;  %??????
114G = [2*ERROR'*Q*THETA zeros(1)];
115P_T=ERROR'*Q*ERROR;
116
117%% 约束条件
118M=tril(ones(Nc));
119Im= eye(Nu);
120M=kron(M,Im);
121
122A_cons=[];
123B_cons=[];
124delta_Umin = kron((-0.2)*ones(Nu,1),ones(Nc,1)); %速度增量约束
125delta_Umax = kron( (0.2)*ones(Nu,1),ones(Nc,1));
126
127lb=[delta_Umin;-5];%（求解方程）状态量下界，包含控制时域内控制增量和松弛因子
128ub=[delta_Umax;5];%（求解方程）状态量上界，包含控制时域内控制增量和松弛因子
129
130%% 以下为约束生成区域
131 %不等式约束
132    A_t=zeros(Nc,Nc);%见falcone论文 P181
133    for p=1:1:Nc
134        for q=1:1:Nc
135            if q<=p
136                A_t(p,q)=1;
137            else
138                A_t(p,q)=0;
139            end
140        end
141    end
142    A_I=kron(A_t,eye(Nu));%对应于falcone论文约束处理的矩阵A,求克罗内克积
143    Ut=kron(ones(Nc,1),u_out);%此处感觉论文里的克罗内科积有问题,暂时交换顺序
144    umin=[-1.6667;-1.6667];%维数与控制变量的个数相同-1.6667
145    umax=[ 1.6667; 1.6667];
146    delta_umin=[-0.2;-0.2;];
147    delta_umax=[0.2;0.2];
148    Umin=kron(ones(Nc,1),umin);
149    Umax=kron(ones(Nc,1),umax);
150    A_cons_cell={A_I zeros(Nu*Nc,1);-A_I zeros(Nu*Nc,1)};
151    b_cons_cell={Umax-Ut;-Umin+Ut};
152    A_cons=cell2mat(A_cons_cell);%（求解方程）状态量不等式约束增益矩阵，转换为绝对值的取值范围
153    B_cons=cell2mat(b_cons_cell);%（求解方程）状态量不等式约束的取值
154   % 状态量约束
155    M=10; 
156    delta_Umin=kron(ones(Nc,1),delta_umin);
157    delta_Umax=kron(ones(Nc,1),delta_umax);
158    lb=[delta_Umin;0];%（求解方程）状态量下界，包含控制时域内控制增量和松弛因子
159    ub=[delta_Umax;M];%（求解方程）状态量上界，包含控制时域内控制增量和松弛因子
160    
161
162

164coder.extrinsic('quadprog');
165
166[DELTA_U,fval]=quadprog(H,G,A_cons,B_cons,[],[],lb,ub)

169u_out = delta_u(1:2)+u_out
