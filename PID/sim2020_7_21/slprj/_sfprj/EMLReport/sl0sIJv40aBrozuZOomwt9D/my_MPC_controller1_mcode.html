1% 最近修改时间：3.1
  2% 基于运动学模型的MPC控制器
  3% u_out=[vel_l ; vel_r]
  4% ref_vel=[xr yr alphar vr omegar]
  5% e = [ex ey ealpha]
  6
  7
  8function [u_out, e_pos, e_vel]  = my_MPC_controller(ref_pos,ref_vel,statu,u_out)
  9%% 底盘参数
 10d = 0.18;
 11l = 0.16;
 12m = 24;
 13Iz = 11.89;
 14c = 513.6; %轮胎侧偏刚度
 15
 16%% 状态量
 17                                                                             %   statu=[ 0 0 0 0 0 0]
 18x = statu(1);
 19y = statu(2);
 20alpha = statu(3);
 21vx = statu(4);
 22vy = statu(5);
 23omega = statu(6);
 24
 25Nc= 25; Np=50; %预测时域Np	控制时域Nc
 26
 27T=0.05;
 28% omega_l=u(1);  % omega_r=u(2);                                          
 29                                                                            % ref_pos=[1 1 0]; ref_vel=[2 0 0];      
 30Xr=ref_pos(1);  Yr=ref_pos(2);  ALPHAr=ref_pos(3);
 31vxr=ref_vel(1); vyr=ref_vel(2); omegar=ref_vel(3);
 32
 33% ur=[90*pi/180;90*pi/180];
 34
 35% v_x = body_v(1)*cos(ALPHA_I)+body_v(2)*sin(ALPHA_I);
 36% v_y = -body_v(1)*sin(ALPHA_I)+body_v(2)*cos(ALPHA_I);
 37% beta_body = atan(v_y/(v_x+0.0000000001));   %质心处beta角
 38% v_x=body_v_b(1);
 39% v_y=body_v_b(2);
 40% beta_body=atan(v_y/(v_x+0.0000000001));   %质心处beta角
 41                                                                                    %    u_out=[0; 0]
 42%控制量个数Nu
 43Nu=size(u_out,1);
 44
 45% s_l = (omega_l*wheel_radius-(v_x-omega*d/2))/(omega_l*wheel_radius+0.000001)
 46% s_r = (omega_l*wheel_radius-(v_x+omega*d/2))/(omega_r*wheel_radius+0.000001)
 47% s_l = (vel_l-(v_x-omega*d))/(vel_l+0.000000000001);
 48% s_r = (vel_l-(v_x+omega*d))/(vel_r+0.000000000001);
 49% s_l = 1-( ( dot_xI+omega*l*cos(ALPHA_I) )/( vel_r*cos(ALPHA_I)+0.00001 ) );
 50% s_r = 1-( ( dot_xI-omega*l*cos(ALPHA_I) )/( vel_l*cos(ALPHA_I)+0.00001 ) );
 51
 52% s=[s_l,s_r];
 53
 54
 55% 
 56% R_B2I=[cos(ALPHA_I-beta_body)  0;
 57%        sin(ALPHA_I-beta_body)  0;
 58%        0                       1];
 59% % omega2vel=(1/2).*[(1-s_l)/cos(beta_body)    (1-s_r)/cos(beta_body);  %[v_l ,v_r]->[v,omega]
 60% %                             -(1-s_l)/d                 (1-s_r)/d ];
 61% omega2vel=(1/2).*[1/cos(beta_body)    1/cos(beta_body);  %[v_l ,v_r]->[v,omega]
 62%                         -1/d                 1/d ];
 63% 
 64% ur=omega2vel\vref;
 65% 
 66% vel_lr=ur(1);
 67% vel_rr=ur(2);
 68
 69%状态量个数Nx
 70statu_error=[x-Xr;   y-Yr;   alpha-ALPHAr;  vx-vxr; vy-vyr; omega-omegar]; 
 71Nx=size(statu_error,1);
 72
 73%  V=R_B2I*omega2vel*u;%运动学模型
 74
 75% 控制矩阵求解(将连续矩阵转换为离散矩阵，采用近似算法  A=I+T*A(t),B=T*B(t))
 76% 此处：delta_X=AX+BU ————X=[a-r]=[X_I-Xr;Y_I-Yr;ALPHA_I-ALPHAr]statu_e=A*statu_e+B*u_e   %离散后的模型
 77    % OUT=delta_X=a-r
 78A=[ 1, 0, (-vx*sin(alpha)-vy*cos(alpha))*T, T*cos(alpha), T*sin(alpha), 0;
 79    0, 1, (vx*cos(alpha)-vy*sin(alpha))*T,  T*sin(alpha), T*cos(alpha), 0;
 80    0, 0, 1, 0, 0, T;
 81    0, 0, 0, 1, 0, 0;
 82    0, 0, 0, T*(2*c*(vy+l*omega))/(m*vx^2+0.0001), (-2*c*T)/(m*vx+0.0001)+1, (-2*c*l*T)/(m*vx+0.0001);
 83    0, 0, 0, -T*(2*c*l*(vy+l*omega))/(Iz*vx^2+0.0001), 2*c*l*T/(Iz*vx+0.0001), (2*c*l*l*T)/(vx*Iz+0.0001)+1;]  %+0.0001是为了避免分母为零
 84
 85B=[0 0;
 86   0 0;
 87   0 0;
 88   T/m T/m;
 89   0 0;
 90   -T*d/Iz T*d/Iz;]
 91
 92C = eye(Nx);
 93
 94Error = C * statu_error; % 误差
 95
 96e_pos=Error(1:Nx/2); %位置误差
 97e_vel=Error(Nx/2:Nx); %速度误差
 98
 99% 得到新的X=A_new*X+B_new*delta_u   X=[X;u-ur]
100A_new = [A B;zeros(Nu,Nx) eye(Nu)];
101B_new = [B;eye(Nu)];
102C_new = [C,zeros(Nx,Nu)];
103%% 
104ur=[0; 0];
105
106statu_new = [Error;u_out-ur];
107
108PHI_cell=cell(Np,1);
109THETA_cell=cell(Np,Nc);
110    for i=1:1:Np
111        PHI_cell{i,1}=C_new*A_new^i;
112        for j=1:1:Nc
113            if j<=i
114                THETA_cell{i,j}=C_new*A_new^(i-j)*B_new;
115            else 
116                THETA_cell{i,j}=zeros(Nx,Nu);
117            end
118        end
119    end
120 PHI=cell2mat(PHI_cell);
121 THETA=cell2mat(THETA_cell);
122
123q = [1 0 0 0 0 0;
124     0 1 0 0 0 0;
125     0 0 1 0 0 0;
126     0 0 0 1 0 0;
127     0 0 0 0 1 0;
128     0 0 0 0 0 1;];
129Q_cell = cell(Np,Np);
130for i=1:1:Np
131    for j=1:1:Np
132        if i==j
133            Q_cell{i,j}=q;
134        else 
135            Q_cell{i,j}=zeros(Nx,Nx);
136        end 
137    end
138end
139Q = cell2mat( Q_cell );
140R = 1*eye(Nu*Nc,Nu*Nc);
141%% 二次型
142p=5;
143AAAA=THETA'*Q*THETA
144H = [THETA'*Q*THETA+R zeros(Nu*Nc,1);zeros(1,Nu*Nc) p];
145ERROR =PHI*statu_new;  %??????
146G = [2*ERROR'*Q*THETA zeros(1)];
147P_T=ERROR'*Q*ERROR;
148
149%% 约束条件
150M=tril(ones(Nc));
151Im= eye(Nu);
152M=kron(M,Im);
153
154A_cons=[];
155B_cons=[];
156delta_Umin = kron((-0.2)*ones(Nu,1),ones(Nc,1)); %速度增量约束
157delta_Umax = kron( (0.2)*ones(Nu,1),ones(Nc,1));
158
159lb=[delta_Umin;-5];%（求解方程）状态量下界，包含控制时域内控制增量和松弛因子
160ub=[delta_Umax;5];%（求解方程）状态量上界，包含控制时域内控制增量和松弛因子
161
162%% 以下为约束生成区域
163 %不等式约束
164    A_t=zeros(Nc,Nc);%见falcone论文 P181
165    for p=1:1:Nc
166        for q=1:1:Nc
167            if q<=p
168                A_t(p,q)=1;
169            else
170                A_t(p,q)=0;
171            end
172        end
173    end
174    A_I=kron(A_t,eye(Nu));%对应于falcone论文约束处理的矩阵A,求克罗内克积
175    Ut=kron(ones(Nc,1),u_out);%此处感觉论文里的克罗内科积有问题,暂时交换顺序
176    umin=[-1.6667;-1.6667];%维数与控制变量的个数相同-1.6667
177    umax=[ 1.6667; 1.6667];
178    delta_umin=[-0.2;-0.2;];
179    delta_umax=[0.2;0.2];
180    Umin=kron(ones(Nc,1),umin);
181    Umax=kron(ones(Nc,1),umax);
182    A_cons_cell={A_I zeros(Nu*Nc,1);-A_I zeros(Nu*Nc,1)};
183    b_cons_cell={Umax-Ut;-Umin+Ut};
184    A_cons=cell2mat(A_cons_cell);%（求解方程）状态量不等式约束增益矩阵，转换为绝对值的取值范围
185    B_cons=cell2mat(b_cons_cell);%（求解方程）状态量不等式约束的取值
186   % 状态量约束
187    M=10; 
188    delta_Umin=kron(ones(Nc,1),delta_umin);
189    delta_Umax=kron(ones(Nc,1),delta_umax);
190    lb=[delta_Umin;0];%（求解方程）状态量下界，包含控制时域内控制增量和松弛因子
191    ub=[delta_Umax;M];%（求解方程）状态量上界，包含控制时域内控制增量和松弛因子
192    
193
194
195%%
196
197[DELTA_U,fval]=quadprog(H,G,A_cons,B_cons,[],[],lb,ub);
198
199u_out=DELTA_U(1:2)+u_out
200
201

205% vyr=ref_vel(2);
206% omegar=ref_vel(3);

214% %

218% % 
219% k1=3;

221% k3=3;
222% 
223%

227% e_alpha=e(3);
228% 
229% vx=vxr*cos(e_alpha)+k1*e_x;
230% vy=0;
231% omega=omegar+k2*vxr*e_y+k3*vxr*sin(e_alpha);
232% % omega=omegar+k2*vxr*sin(e_alpha)/e_alpha*e_y+k3*e_alpha;
233% 
234% v=[vx;vy;omega]
235% % u=[1;0;1]
236end
