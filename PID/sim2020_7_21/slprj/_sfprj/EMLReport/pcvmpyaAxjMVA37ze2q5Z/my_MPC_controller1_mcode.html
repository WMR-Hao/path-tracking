1function [u_out, e_pos, e_vel] = my_MPC_controller(ref_pos, ref_vel, statu, u_out)
  2% 最近修改时间：7.25 
  3% 基于运动学模型的MPC控制器
  4% u_out=[vx ;  omega]
  5% ref_vel=[xr yr alphar vr omegar]
  6% e = [ex ey ealpha]
  7
  8Nc= 2; Np=4; %预测时域Np	控制时域Nc
  9T=0.05; 
 10
 11% 测试是否能得到
 12%    statu=[2; -2; pi/2; 0.001; 0.001; 0.001];
 13%    u_out=[0; 0];
 14%    ref_pos=[0;0;pi/4]; ref_vel=[2; 0; 0];  
 15%% 底盘参数
 16d = 0.18;	l = 0.16;	m = 24;	Iz = 11.89;	c = 513.6; %轮胎侧偏刚度
 17
 18%% 状态量                                                                           
 19x = statu(1);   y = statu(2);  alpha = statu(3); 
 20vx = statu(4);  vy = statu(5);    omega = statu(6);
 21                                      
 22                                                                            
 23xr=ref_pos(1);  yr=ref_pos(2);  alphar=ref_pos(3);
 24vxr=ref_vel(1); vyr=ref_vel(2); omegar=ref_vel(3);
 25                                                                                      
 26%控制量个数Nu
 27Nu=size(u_out,1);
 28
 29%状态量个数Nx
 30statu_error=[x-xr;   y-yr;   alpha-alphar;]; 
 31Nx=size(statu_error,1);
 32
 33% 控制矩阵求解(将连续矩阵转换为离散矩阵，采用近似算法  A=I+T*A(t),B=T*B(t))
 34% 此处：delta_X=AX+BU ————X=[a-r]=[X_I-Xr;Y_I-Yr;ALPHA_I-ALPHAr]statu_e=A*statu_e+B*u_e   %离散后的模型
 35    % OUT=delta_X=a-r
 36A=[ 1 0 -vx*sin(alphar)*T; 0 1 vx*cos(alphar)*T; 0 0 1];
 37B=[cos(alphar)*T 0 ; sin(alphar)*T 0 ; 0 T];
 38
 39C = eye(Nx);
 40
 41Error = C * statu_error; % 误差
 42
 43% e_pos=Error(1:Nx/2); %位置误差
 44
 45% 得到新的X=A_new*X+B_new*delta_u   X=[X;u-ur]
 46A_new = [A B;zeros(Nu,Nx) eye(Nu)];
 47B_new = [B;eye(Nu)];
 48C_new = [C,zeros(Nx,Nu)];
 49%% 
 50ur=[vxr; omegar];
 51
 52statu_new = [Error;u_out-ur];
 53
 54PHI=zeros(Np*size(C_new*A_new,1),size(A_new,2));
 55for i=1:1:Np
 56    PHI( Nx*i-(Nx-1):Nx*i  ,:)=C_new*A_new^i;
 57end
 58
 59THETA=zeros( Np*size(C_new*B_new,1),  Nc*size(C_new*A_new*B_new ,2) );
 60for i=1:1:Np
 61    for j=1:1:Nc
 62        if j<= i
 63            THETA( Nx*(i-1)+1:Nx*i,Nu*j-1:Nu*j )= C_new*A_new^(i-j)*B_new;
 64        else
 65            THETA( Nx*(i-1)+1:Nx*i,Nu*j-1:Nu*j ) = zeros( Nx,Nu );
 66        end
 67    end
 68end
 69
 70
 71q = [1 0 0;
 72     0 1 0;
 73     0 0 1;];
 74Q=zeros(Nx*Np,Nx*Np);
 75for i=1:Np
 76    for j=1:Np
 77        if i==j
 78            Q(Nx*(i-1)+1:Nx*i , Nx*(i-1)+1:Nx*i) = q;
 79        end
 80    end
 81end
 82
 83R = 1*eye( Nc*Nu,Nc*Nu);
 84
 85%% 二次型
 86p=5;
 87
 88H = [2*THETA'*Q*THETA+R zeros(Nc*Nu,1);zeros(1,Nc*Nu) p];  %不是对称的 % 前面加了个2
 89
 90ERROR =PHI*statu_new;  
 91G = [2*ERROR'*Q*THETA zeros(1)];
 92P_T = ERROR'*Q*ERROR;
 93
 94%% 约束条件
 95%控制量约束
 96M=tril(ones(Nc));
 97Im= eye(Nu);
 98A=kron(M,Im); % U=A*DELTA_U+Ut  中的A
 99
100Ut=kron(ones(Nc,1),u_out); %Ut
101
102umin=[-5;-2];  %维数与控制变量的个数相同
103umax=[ 5; 2];
104Umin=kron(ones(Nc,1),umin);
105Umax=kron(ones(Nc,1),umax);
106a = [A zeros(Nu*Nc,1);-A zeros(Nu*Nc,1)];
107b = [Umax-Ut;-Umin+Ut];
108
109 % 状态增量约束
110delta_u_range=2;
111delta_Umin = kron((-delta_u_range)*ones(Nu,1),ones(Nc,1)); %速度增量约束
112delta_Umax = kron( (delta_u_range)*ones(Nu,1),ones(Nc,1));
113
114lb=[delta_Umin;-10];%（求解方程）状态量下界，包含控制时域内控制增量和松弛因子
115ub=[delta_Umax;10];%（求解方程）状态量上界，包含控制时域内控制增量和松弛因子
116  
117%%
118coder.extrinsic('quadprog');
119coder.extrinsic('tic');
120coder.extrinsic('toc');
121tic 
122[DELTA_U,fval]=quadprog(H,G,a,b,[],[],lb,ub);
123toc
124% DELTA_U=quadprog(H,G,[],[],[],[],lb,ub)
125delta_u = zeros(Nu*Nc+1,1);	% 定义类型，存储DELTA_U的mxarray类型数据
126delta_u = DELTA_U;
127delta_uout = delta_u(1:2)
128u_out = delta_uout+u_out
129
